```css

    --spacing-sm: 0.5rem
    --spacing-md: 1rem
    --spacing-lg: 1.5rem

    --radius-md: 0.5rem
    --radius-lg: 0.75rem
    --radius-xl: 1rem
    --radius-full: 9999px

    --shadow-md: 0 4px 6px rgba(0,0,0,0.1)
    --shadow-lg: 0 10px 15px rgba(0,0,0,0.1)
    --shadow-xl: 0 20px 25px rgba(0,0,0,0.1)

    --transition-base: 250ms cubic-bezier(0.4,0,0.2,1)
    --transition-slow: 350ms cubic-bezier(0.4,0,0.2,1)
}
```
